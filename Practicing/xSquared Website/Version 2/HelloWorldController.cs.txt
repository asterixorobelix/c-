using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.Mvc;

namespace Squared.Controllers
{
    public class HelloWorldController : Controller
    {
        // GET: HelloWorld
        public string Index()
        {
            return "This page will double the number which you give it.";
        }

        // 
        // GET: /HelloWorld/Welcome/ 

        public string Welcome(string name = "Nathan", int ID=1)
        {
            return HttpUtility.HtmlEncode("Hello " +name+", "+ID+" doubled is: "+ID*ID); // http://localhost:xxx/HelloWorld/Welcome/3?name=Rick
            //Here, we are passing ID as route data and name as a query string. Route data is the more typical way to do it.
        }
    }
}
