using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.Mvc;
using System.Web.Routing;

namespace Squared
{
    public class RouteConfig
    {
        public static void RegisterRoutes(RouteCollection routes)
        {
            routes.IgnoreRoute("{resource}.axd/{*pathInfo}");

            routes.MapRoute(
                name: "Default",
                url: "{controller}/{action}/{id}",
                defaults: new { controller = "Home", action = "Index", id = UrlParameter.Optional }
            );

            //added a route to pass both the name and numtimes in parameters as route data in the URL
            routes.MapRoute(
                name: "HelloWorld",
                url: "{controller}/{action}/{name}/{id}" //specifies that any URL that starts with domainName/HelloWorld, must be handled by HelloWorldController
                                                         // localhost:XXX/HelloWorld/Welcome/Scott/3
                );
        }
    }
}
