using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.Mvc;

namespace Squared.Controllers
{
    public class HelloWorldController : Controller
    {
        // GET: HelloWorld
        public ActionResult Index() //needs to return an ActionResult, rather than a string, if its returning a View();
        {   //The Index method above uses a view template to generate an HTML response to the browser
            return View();
        }

        // 
        // GET: /HelloWorld/Welcome/ 

        public string Welcome(string name = "Nathan", int ID=1)
        {
            return HttpUtility.HtmlEncode("Hello " +name+", "+ID+" doubled is: "+ID*ID); // localhost:XXX/HelloWorld/Welcome/Scott/3
            //Here, we are passing both ID and name as route data.
		//This system also ouputs the values directly to the browser, rather than passing it to the View().
        }
    }
}
